 66:       v_len = scores_old.vsize
 67:       p_len = scores_old.hsize
 68: 
 69:       v = DMatrix.new(v_len, 1, 0.5)
 70:       p = DMatrix.new(1, p_len, 0.1)
 71: 
 72:       old_num_errs = [Inf] * 10
 73:       old_err = [Inf] * 10
 74:       stop = false
 75: 
 76:       until stop do
 77:         scores = v * p
 78:         dv = DMatrix.new(v.vsize, v.hsize, 0)
 79:         dp = DMatrix.new(p.vsize, p.hsize, 0)
 80: 
 81:         err = 0
 82:         num_errs = 0
 83: 
 84:         for i in 0...votes.length
 85:           for winner, loser in votes[i]
 86:             winner_val = scores_old[i, winner] + scores[i, winner]
 87:             loser_val = scores_old[i, loser] + scores[i, loser]
 88: 
 89:             e = winner_val - loser_val
 90:             num_errs -= 1 if e > 0
 91:             e = 1 - e
 92:             err += e * e
 93:             num_errs += 1
 94: 
 95:             dv[i] += h*2*e*(p[0,winner] - p[0,loser])
 96:             adj = h*2*e*v[i,0]
 97:             dp[0,loser] -= adj
 98:             dp[0,winner] += adj
 99:           end
100:         end
101: 
102:         puts "[#{num_errs}, #{err}]"
103: 
104:         v += dv
105:         p += dp
106: 
107:         old_num_errs << num_errs
108:         old_err << err
109: 
110:         stop = err_window(old_num_errs) && err_window(old_err)
111:

113:         old_err.shift
